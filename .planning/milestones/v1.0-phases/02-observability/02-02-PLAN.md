1. Rewrite `src/Starter.WebApi/Program.cs` to this exact structure:

```csharp
using Serilog;
using Starter.ExceptionHandling;
using Starter.Logging;

// --- Bootstrap Logger ---
// Lightweight logger for startup and crash capture.
// Replaced by the full Serilog pipeline once AddAppLogging() runs.
Log.Logger = new LoggerConfiguration()
    .MinimumLevel.Information()
    .WriteTo.Console()
    .CreateBootstrapLogger();

try
{
    Log.Information("Starting application");

    var builder = WebApplication.CreateBuilder(args);

    // --- Observability ---
    builder.AddAppLogging();

    // --- Security ---
    // (Phase 4: Identity + Google OAuth + JWT Bearer)

    // --- Data ---
    // (Phase 3: EF Core + SQLite)

    // --- API ---
    builder.Services.AddControllers();
    builder.Services.AddAppExceptionHandling();

    var app = builder.Build();

    // --- Middleware Pipeline ---
    app.UseAppExceptionHandling(); // Must be first
    app.UseHttpsRedirection();
    app.UseAppRequestLogging();    // After exception handler and HTTPS redirect

    // (Phase 4: app.UseAuthentication(), app.UseAuthorization())

    app.MapControllers();

    app.Run();
}
catch (Exception ex)
{
    Log.Fatal(ex, "Application terminated unexpectedly");
}
finally
{
    Log.CloseAndFlush();
}
```

Key points:
- `using Serilog;` is required for `Log.Logger`, `Log.Information`, `Log.Fatal`, `Log.CloseAndFlush`
- `using Starter.Logging;` is required for `AddAppLogging()` and `UseAppRequestLogging()` extension methods
- The bootstrap logger and try/catch/finally MUST live in Program.cs (NOT inside the module) because they need to exist before any module loads and after everything shuts down (anti-pattern from research)
- `UseAppRequestLogging()` is placed AFTER `UseAppExceptionHandling()` and `UseHttpsRedirection()` (middleware ordering from research anti-patterns)
- The Phase 4 comments are moved inside the try block after `UseAppRequestLogging()`

```json
// Configuration Guidance:
//   Development: Use User Secrets (dotnet user-secrets set "Section:Key" "value")
//   Environment Variables: Use double-underscore for nested keys (Serilog__MinimumLevel__Default=Warning)

{

    "IncludeStackTraceInDevelopment": true
  },

  "Serilog": {
    "MinimumLevel": {
      "Default": "Information",
      "Override": {
        "Microsoft": "Warning",
        "Microsoft.AspNetCore": "Warning",
        "Microsoft.Hosting.Lifetime": "Information",
        "System": "Warning"
      }
    },
    "Enrich": [
      "FromLogContext",
      "WithEnvironmentName",
      "WithMachineName",
      {
        "Name": "WithCorrelationId",
        "Args": {
          "headerName": "X-Correlation-Id",
          "addValueIfHeaderAbsence": true
        }
      }
    ],
    "Properties": {
      "Application": "Starter.WebApi"
    },
    "Sinks": {
      "Console": {
        "Enabled": true,
        "OutputTemplate": "[{Timestamp:HH:mm:ss} {Level:u3}] {Message:lj}{NewLine}{Exception}"
      },
      "File": {
        "Enabled": false,
        // To enable file logging, set Enabled to true.
        // Logs are written to the Path with daily rolling files.
        "Path": "Logs/log-.txt",
        "RollingInterval": "Day",
        "RetainedFileCountLimit": 31,
        "FileSizeLimitBytes": 104857600
      },
      "Seq": {
        "Enabled": false,
        // To enable Seq, set Enabled to true and install Seq (https://datalust.co/seq).
        // Seq provides a structured log viewer for local development.
        "ServerUrl": "http://localhost:5341"
      },
      "OpenTelemetry": {
        "Enabled": false,
        // For generic OTLP collector: set Endpoint to your collector URL (e.g., http://localhost:4317)
        // For Azure Application Insights: set Endpoint to your ingestion URL
        //   (e.g., https://eastus-8.in.applicationinsights.azure.com/) and Protocol to "HttpProtobuf"
        "Endpoint": "http://localhost:4317",
        "Protocol": "Grpc"
      }
    },
    "RequestLogging": {
      "EnableClientIp": true,
      "EnableUserAgent": true,
      "EnableContentType": true
    }
  }
}
```

NOTE: ASP.NET Core supports lenient JSON with comments (already established in Phase 1 appsettings.json). The JSON comments above are intentional guidance for users.
